    <script>

        const PARTIDAS = [

            { id: 'alba', nombre: 'Albañilería', p_venta: 45 },

            { id: 'inst', nombre: 'Instalaciones', p_venta: 75 },

            { id: 'reves', nombre: 'Revestimientos', p_venta: 55 }

        function init() {

            cargarDatos();

        }

        function renderTabla() {

            const tbody = document.getElementById('tablaBalance');

            tbody.innerHTML = '';

           

            PARTIDAS.forEach(p => {

                const qty = localStorage.getItem(`qty-${p.id}`) || 0;

                const real = localStorage.getItem(`real-${p.id}`) || 0;

                const pptado = qty * p.p_venta;

                const balance = pptado - real;

                const balanceClase = balance >= 0 ? 'desviacion-ok' : 'desviacion-peligro';

                tbody.innerHTML += `

                    <tr class="border-t border-slate-100">

                        <td class="p-4 font-bold text-slate-700">${p.nombre}</td>

                        <td class="p-4">

                            <input type="number" value="${qty}" onchange="save('qty-${p.id}', this.value)" class="w-16 border rounded p-1">

                        </td>

                        <td class="p-4">

                            <input type="number" value="${real}" onchange="save('real-${p.id}', this.value)" class="w-16 border rounded p-1 bg-amber-50">

                        </td>

                        <td class="p-4 ${balanceClase}">${balance.toFixed(2)} €</td>

                    </tr>

                `;

            });

            recalcularGlobal();

        }

        function save(key, val) {

            localStorage.setItem(key, val);

            renderTabla();

        }

        function recalcularGlobal() {

            let totalVenta = 0;

            let totalGasto = 0;

           

            PARTIDAS.forEach(p => {

                const qty = localStorage.getItem(`qty-${p.id}`) || 0;

                const real = localStorage.getItem(`real-${p.id}`) || 0;

                totalVenta += (qty * p.p_venta);

                totalGasto += parseFloat(real);

            });

            const totalConIva = totalVenta * 1.10;

            const beneficio = totalVenta - totalGasto;

            document.getElementById('totalFinal').innerText = totalConIva.toFixed(2) + " €";

            document.getElementById('gastoTotalReal').innerText = totalGasto.toFixed(2) + " €";

            document.getElementById('beneficioNeto').innerText = beneficio.toFixed(2) + " €";

            document.getElementById('margenTotal').innerText = beneficio.toFixed(2) + " €";

        }

        // --- LÓGICA DE FIRMA ---

        function initCanvas() {

            const canvas = document.getElementById('canvasFirma');

            const ctx = canvas.getContext('2d');

            let drawing = false;

            canvas.addEventListener('mousedown', () => drawing = true);

            canvas.addEventListener('mouseup', () => { drawing = false; ctx.beginPath(); });

            canvas.addEventListener('mousemove', draw);

            // Soporte táctil

            canvas.addEventListener('touchstart', (e) => { drawing = true; e.preventDefault(); });

            canvas.addEventListener('touchend', () => { drawing = false; ctx.beginPath(); });

            canvas.addEventListener('touchmove', (e) => {

                const touch = e.touches[0];

                const rect = canvas.getBoundingClientRect();

                draw({ clientX: touch.clientX, clientY: touch.clientY });

                e.preventDefault();

            });

            function draw(e) {

                if (!drawing) return;

                const rect = canvas.getBoundingClientRect();

                ctx.lineWidth = 2;

                ctx.lineCap = 'round';

                ctx.strokeStyle = '#1e293b';

                ctx.lineTo(e.clientX - rect.left, e.clientY - rect.top);

                ctx.stroke();

                ctx.beginPath();

                ctx.moveTo(e.clientX - rect.left, e.clientY - rect.top);

                firmaData = canvas.toDataURL();

            }

        }

        function limpiarFirma() {

            const canvas = document.getElementById('canvasFirma');

            canvas.getContext('2d').clearRect(0, 0, canvas.width, canvas.height);

            firmaData = null;

        }

        function generarPDFPro() {

            const { jsPDF } = window.jspdf;

            const doc = new jsPDF();

           

            doc.setFontSize(20);

            doc.text("CONTRATO DE REFORMA Y LIQUIDACIÓN", 20, 20);

           

            // Tabla de partidas

            const body = PARTIDAS.map(p => {

                const qty = localStorage.getItem(`qty-${p.id}`) || 0;

                return [p.nombre, qty, (qty * p.p_venta).toFixed(2) + " €"];

            });

            doc.autoTable({

                startY: 30,

                head: [['Partida', 'Cantidad', 'Subtotal']],

                body: body

            });

            // Añadir firma si existe

            if(firmaData) {

                const finalY = doc.lastAutoTable.finalY + 20;

                doc.text("Firma del Cliente:", 20, finalY);

                doc.addImage(firmaData, 'PNG', 20, finalY + 5, 50, 20);

            }

            doc.save("Contrato_ConstruCalc_Elite.pdf");

        }
